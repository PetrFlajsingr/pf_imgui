40  void addRow(const std::ranges::range auto &vals) {

48  void removeRow(std::size_t index) {
49  const auto startIndex = index * ColumnCount;
50  if (startIndex >= rows.size()) { return; }
51  rows.erase(rows.begin() + startIndex, rows.begin() + startIndex + ColumnCount);
52  }
53  [[nodiscard]] std::span<std::string> getRow(std::size_t index) {
54  const auto startIndex = index * ColumnCount;
55  if (startIndex >= rows.size()) { return {}; }
56  return {rows.begin() + startIndex, rows.begin() + startIndex + ColumnCount};
57  }
58  [[nodiscard]] std::string &getCell(std::size_t rowIdx, std::size_t columnIdx) {
59  auto row = getRow(rowIdx);
60  assert(row.size() < columnIdx);
61  return row[columnIdx];
62  }
