255  if constexpr (std::derived_from<T, details::DropTargetBase>) {

259  std::ranges::remove_if(sources, [&arg1](const auto &s) { return s == &arg1; });

261  if constexpr (sizeof...(Args) > 0) { remove(std::forward<Args &>(args)...); }
262  }
