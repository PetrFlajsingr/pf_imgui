63 static inline bool ComboFilter__DrawPopup(ComboFilterState &state, [[maybe_unused]] int START, const char **ENTRIES,
64  int ENTRY_COUNT) {
65  bool clicked = 0;
66 
67  // Grab the position for the popup
68  ImVec2 pos = ImGui::GetItemRectMin();
69  pos.y += ImGui::GetItemRectSize().y;
70  ImVec2 size = ImVec2(ImGui::GetItemRectSize().x - 60, ImGui::GetFrameHeightWithSpacing() * 4);
71 
72  ImGui::PushStyleVar(ImGuiStyleVar_WindowRounding, 0);
73 
74  ImGuiWindowFlags flags = ImGuiWindowFlags_NoTitleBar | ImGuiWindowFlags_NoResize | ImGuiWindowFlags_NoMove
75  | ImGuiWindowFlags_HorizontalScrollbar | ImGuiWindowFlags_NoSavedSettings | 0;//ImGuiWindowFlags_ShowBorders;
76 
77  ImGui::SetNextWindowFocus();
78 
79  ImGui::SetNextWindowPos(pos);
80  ImGui::SetNextWindowSize(size);
81  ImGui::Begin("##combo_filter", nullptr, flags);
82 
83  ImGui::PushAllowKeyboardFocus(false);
84 
85  for (int i = 0; i < ENTRY_COUNT; i++) {
86  // Track if we're drawing the active index so we
87  // can scroll to it if it has changed
88  bool isIndexActive = state.activeIdx == i;
89 
90  if (isIndexActive) {
91  // Draw the currently 'active' item differently
92  // ( used appropriate colors for your own style )
93  ImGui::PushStyleColor(ImGuiCol_Border, ImVec4(1, 1, 0, 1));
94  }
95 
96  ImGui::PushID(i);
97  if (ImGui::Selectable(ENTRIES[i], isIndexActive)) {
98  // And item was clicked, notify the input
99  // callback so that it can modify the input buffer
100  state.activeIdx = i;
101  clicked = 1;
102  }
103  if (ImGui::IsItemFocused() && ImGui::IsKeyPressed(ImGui::GetIO().KeyMap[ImGuiKey_Enter])) {
104  // Allow ENTER key to select current highlighted item (w/ keyboard navigation)
105  state.activeIdx = i;
106  clicked = 1;
107  }
108  ImGui::PopID();
109 
110  if (isIndexActive) {
111  if (state.selectionChanged) {
112  // Make sure we bring the currently 'active' item into view.
113  ImGui::SetScrollHereX();
114  ImGui::SetScrollHereY();
115  state.selectionChanged = false;
116  }
117 
118  ImGui::PopStyleColor(1);
119  }
120  }
121 
122  ImGui::PopAllowKeyboardFocus();
123  ImGui::End();
124  ImGui::PopStyleVar(1);
125 
126  return clicked;
127 }

129 static inline bool ComboFilter(const char *id, char *buffer, int bufferlen, const char **hints, int num_hints,
130  ComboFilterState &s) {
131  struct fuzzy {
132  static int score(const char *str1, const char *str2) {
133  int score = 0, consecutive = 0, maxerrors = 0;
134  while (*str1 && *str2) {
135  int is_leading = (*str1 & 64) && !(str1[1] & 64);
136  if ((*str1 & ~32) == (*str2 & ~32)) {
137  int had_separator = (str1[-1] <= 32);
138  int x = had_separator || is_leading ? 10 : consecutive * 5;
139  consecutive = 1;
140  score += x;
141  ++str2;
142  } else {
143  int x = -1, y = is_leading * -3;
144  consecutive = 0;
145  score += x;
146  maxerrors += y;
147  }
148  ++str1;
149  }
150  return score + (maxerrors < -9 ? -9 : maxerrors);
151  }
152  static int search(const char *str, int num, const char *words[]) {
153  int scoremax = 0;
154  int best = -1;
155  for (int i = 0; i < num; ++i) {
156  int score = fuzzy::score(words[i], str);
157  int record = (score >= scoremax);
158  int draw = (score == scoremax);
159  if (record) {
160  scoremax = score;
161  if (!draw) best = i;
162  else
163  best = best >= 0 && strlen(words[best]) < strlen(words[i]) ? best : i;
164  }
165  }
166  return best;
167  }
168  };
169  bool done =
170  ImGui::InputText(id, buffer, bufferlen, ImGuiInputTextFlags_AutoSelectAll | ImGuiInputTextFlags_EnterReturnsTrue);
171  bool hot = s.activeIdx >= 0 && strcmp(buffer, hints[s.activeIdx]);
172  if (hot) {
173  int new_idx = fuzzy::search(buffer, num_hints, hints);
174  int idx = new_idx >= 0 ? new_idx : s.activeIdx;
175  s.selectionChanged = s.activeIdx != idx;
176  s.activeIdx = idx;
177  if (done || ComboFilter__DrawPopup(s, idx, hints, num_hints)) {
178  int i = s.activeIdx;
179  if (i >= 0) {
180  strcpy(buffer, hints[i]);
181  done = true;
182  }
183  }
184  }
185  return done;
186 }
