13 IMGUI_API inline bool SpinScaler(const char* label, ImGuiDataType data_type, void* data_ptr, const void* step, const void* step_fast, const char* format, ImGuiInputTextFlags flags)
14 {
15  ImGuiWindow* window = GetCurrentWindow();
16  if (window->SkipItems)
17  return false;
18 
19  ImGuiContext& g = *GImGui;
20  ImGuiStyle& style = g.Style;
21 
22  if (format == NULL)
23  format = DataTypeGetInfo(data_type)->PrintFmt;
24 
25  char buf[64];
26  DataTypeFormatString(buf, IM_ARRAYSIZE(buf), data_type, data_ptr, format);
27 
28  bool value_changed = false;
29  if ((flags & (ImGuiInputTextFlags_CharsHexadecimal | ImGuiInputTextFlags_CharsScientific)) == 0)
30  flags |= ImGuiInputTextFlags_CharsDecimal;
31  flags |= ImGuiInputTextFlags_AutoSelectAll;
32  flags |= ImGuiInputTextFlags_NoMarkEdited; // We call MarkItemEdited() ourselve by comparing the actual data rather than the string.
33 
34  if (step != NULL)
35  {
36  const float button_size = GetFrameHeight();
37 
38  BeginGroup(); // The only purpose of the group here is to allow the caller to query item data e.g. IsItemActive()
39  PushID(label);
40  SetNextItemWidth(ImMax(1.0f, CalcItemWidth() - (button_size + style.ItemInnerSpacing.x) * 2));
41  if (InputText("", buf, IM_ARRAYSIZE(buf), flags)) // PushId(label) + "" gives us the expected ID from outside point of view
42  value_changed = DataTypeApplyOpFromText(buf, g.InputTextState.InitialTextA.Data, data_type, data_ptr, format);
43 
44  // Step buttons
45  const ImVec2 backup_frame_padding = style.FramePadding;
46  style.FramePadding.x = style.FramePadding.y;
47  ImGuiButtonFlags button_flags = ImGuiButtonFlags_Repeat | ImGuiButtonFlags_DontClosePopups;
48  if (flags & ImGuiInputTextFlags_ReadOnly)
49  button_flags |= ImGuiButtonFlags_Disabled;
50  SameLine(0, style.ItemInnerSpacing.x);
51 
52 // start diffs
53  float frame_height = GetFrameHeight();
54  float arrow_size = std::floor(frame_height * .45f);
55  float arrow_spacing = frame_height - 2.0f * arrow_size;
56 
57  BeginGroup();
58  PushStyleVar(ImGuiStyleVar_ItemSpacing, ImVec2{g.Style.ItemSpacing.x, arrow_spacing});
59 
60  // save/change font size to draw arrow buttons correctly
61  float org_font_size = GetDrawListSharedData()->FontSize;
62  GetDrawListSharedData()->FontSize = arrow_size;
63 
64  if (ArrowButtonEx("+", ImGuiDir_Up, ImVec2(arrow_size, arrow_size), button_flags))
65  {
66  DataTypeApplyOp(data_type, '+', data_ptr, data_ptr, g.IO.KeyCtrl && step_fast ? step_fast : step);
67  value_changed = true;
68  }
69 
70  if (ArrowButtonEx("-", ImGuiDir_Down, ImVec2(arrow_size, arrow_size), button_flags))
71  {
72  DataTypeApplyOp(data_type, '-', data_ptr, data_ptr, g.IO.KeyCtrl && step_fast ? step_fast : step);
73  value_changed = true;
74  }
75 
76  // restore font size
77  GetDrawListSharedData()->FontSize = org_font_size;
78 
79  PopStyleVar(1);
80  EndGroup();
81 // end diffs
82 
83  const char* label_end = FindRenderedTextEnd(label);
84  if (label != label_end)
85  {
86  SameLine(0, style.ItemInnerSpacing.x);
87  TextEx(label, label_end);
88  }
89  style.FramePadding = backup_frame_padding;
90 
91  PopID();
92  EndGroup();
93  }
94  else
95  {
96  if (InputText(label, buf, IM_ARRAYSIZE(buf), flags))
97  value_changed = DataTypeApplyOpFromText(buf, g.InputTextState.InitialTextA.Data, data_type, data_ptr, format);
98  }
99  if (value_changed)
100  MarkItemEdited(window->DC.LastItemId);
101 
102  return value_changed;
103 }

105 IMGUI_API inline bool SpinInt(const char* label, int* v, int step = 1, int step_fast = 100, ImGuiInputTextFlags flags = 0)
106 {
107  // Hexadecimal input provided as a convenience but the flag name is awkward. Typically you'd use InputText() to parse your own data, if you want to handle prefixes.
108  const char* format = (flags & ImGuiInputTextFlags_CharsHexadecimal) ? "%08X" : "%d";
109  return SpinScaler(label, ImGuiDataType_S32, (void*)v, (void*)(step>0 ? &step : NULL), (void*)(step_fast>0 ? &step_fast : NULL), format, flags);
110 }
