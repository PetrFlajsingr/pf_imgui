145 template<ColorChooserType Type, ColorChooserFormat Format>
146 void ColorChooser<Type, Format>::renderImpl() {
147  [[maybe_unused]] auto colorScoped = color.applyScoped();
148  [[maybe_unused]] auto styleScoped = style.applyScoped();
149  [[maybe_unused]] auto fontScoped = font.applyScopedIfNotDefault();
150  auto flags = pickerEnabled ? ImGuiColorEditFlags{} : ImGuiColorEditFlags_NoPicker;
151  auto valueChanged = false;
152  if constexpr (Type == ColorChooserType::Edit) {
153  if constexpr (Format == ColorChooserFormat::RGB) {
154  valueChanged = ImGui::ColorEdit3(label->get().c_str(), glm::value_ptr(valueStorage), flags);
155  } else {
156  valueChanged = ImGui::ColorEdit4(label->get().c_str(), glm::value_ptr(valueStorage), flags);
157  }
158  } else {
159  if constexpr (Format == ColorChooserFormat::RGB) {
160  valueChanged = ImGui::ColorPicker3(label->get().c_str(), glm::value_ptr(valueStorage), flags);
161  } else {
162  valueChanged = ImGui::ColorPicker4(label->get().c_str(), glm::value_ptr(valueStorage), flags);
163  }
164  }
165  DragSource::drag(*chosenColor);
166  if (auto drop = DropTarget::dropAccept(); drop.has_value()) {
167  *chosenColor.modify() = *drop;
168  return;
169  }
170  if (valueChanged) {
171  if constexpr (Format == ColorChooserFormat::RGB) {
172  *chosenColor.modify() = Color::RGB(valueStorage.r, valueStorage.g, valueStorage.b);
173  } else {
174  *chosenColor.modify() = Color::RGB(valueStorage.r, valueStorage.g, valueStorage.b, valueStorage.a);
175  }
176  }
177 }
