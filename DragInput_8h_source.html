240 template<OneOf<PF_IMGUI_DRAG_TYPE_LIST> T>
241 void DragInput<T>::setFromToml(const toml::table &src) {
242  if constexpr (OneOf<T, PF_IMGUI_DRAG_RANGE_TYPE_LIST>) {
243  if (auto newValIter = src.find("value"); newValIter != src.end()) {
244  if (auto newVal = newValIter->second.as_array(); newVal != nullptr) {
245  if (newVal->size() != 2) { return; }
246  auto range = T{};
247  if (auto newRangeStart = newVal->get(0)->value<ParamType>(); newRangeStart.has_value()) {
248  range.start = newRangeStart.value();
249  }
250  if (auto newRangeEnd = newVal->get(1)->value<ParamType>(); newRangeEnd.has_value()) {
251  range.end = newRangeEnd.value();
252  }
253  *value.modify() = range;
254  }
255  }
256  } else if constexpr (OneOf<T, PF_IMGUI_DRAG_GLM_TYPE_LIST>) {
257  if (auto newValIter = src.find("value"); newValIter != src.end()) {
258  if (auto newVal = newValIter->second.as_array(); newVal != nullptr) {
259  const auto vecValue = safeDeserializeGlmVec<T>(*newVal);
260  if (vecValue.has_value()) { *value.modify() = *vecValue; }
261  }
262  }
263  } else {
264  if (auto newValIter = src.find("value"); newValIter != src.end()) {
265  if (auto newVal = newValIter->second.value<T>(); newVal.has_value()) { *value.modify() = *newVal; }
266  }
267  }
